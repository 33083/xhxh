import json
import os
from datetime import datetime, timedelta

DATA_FILE = "library_data.json"

def load_data():
    if os.path.exists(DATA_FILE):
        with open(DATA_FILE, "r", encoding="utf-8") as f:
            return json.load(f)
    return {"books": {}, "members": {}}

def save_data(data):
    with open(DATA_FILE, "w", encoding="utf-8") as f:
        json.dump(data, f, ensure_ascii=False, indent=2)

def add_book(data, isbn, title, author, total_copies=1):
    if isbn in data["books"]:
        print("该书已存在，如需增加副本请使用更新功能。")
        return
    data["books"][isbn] = {
        "title": title,
        "author": author,
        "total_copies": total_copies,
        "available_copies": total_copies,
        "borrowed_by": []
    }
    save_data(data)
    print(f"成功添加《{title}》(ISBN: {isbn})，共{total_copies}册")

def remove_book(data, isbn):
    if isbn not in data["books"]:
        print("未找到该书。")
        return
    book = data["books"][isbn]
    if len(book["borrowed_by"]) > 0:
        print(f"该书还有{len(book['borrowed_by'])}册被借出，无法删除。请先归还所有副本。")
        return
    del data["books"][isbn]
    save_data(data)
    print("已成功删除。")

def update_book(data, isbn, title=None, author=None, total_copies=None):
    if isbn not in data["books"]:
        print("未找到该书。")
        return
    book = data["books"][isbn]
    if title:
        book["title"] = title
    if author:
        book["author"] = author
    if total_copies is not None:
        diff = total_copies - book["total_copies"]
        book["total_copies"] = total_copies
        book["available_copies"] = max(0, book["available_copies"] + diff)
    save_data(data)
    print("图书信息已更新。")

def search_books(data, keyword):
    results = []
    for isbn, book in data["books"].items():
        if keyword.lower() in book["title"].lower() or keyword.lower() in book["author"].lower() or keyword == isbn:
            results.append((isbn, book))
    if not results:
        print("未找到相关图书。")
    else:
        for isbn, book in results:
            status = f"{book['available_copies']}/{book['total_copies']} 可借"
            print(f"ISBN: {isbn} | 书名: {book['title']} | 作者: {book['author']} | {status}")

def borrow_book(data, member_id, isbn):
    if isbn not in data["books"]:
        print("未找到该书。")
        return
    book = data["books"][isbn]
    if book["available_copies"] <= 0:
        print("所有副本都已借出。")
        return

    # 注册成员（简化处理）
    if member_id not in data["members"]:
        data["members"][member_id] = {"borrowed": []}

    # 检查是否已借过此书（避免重复）
    if isbn in data["members"][member_id]["borrowed"]:
        print("您已借过此书。")
        return

    book["available_copies"] -= 1
    book["borrowed_by"].append({"member_id": member_id, "borrow_date": datetime.now().isoformat()})
    data["members"][member_id]["borrowed"].append(isbn)
    save_data(data)
    print(f"借阅成功！《{book['title']}》借给 {member_id}，归还期限为30天。")

def return_book(data, member_id, isbn):
    if isbn not in data["books"]:
        print("未找到该书。")
        return
    book = data["books"][isbn]
    member = data["members"].get(member_id)
    if not member or isbn not in member["borrowed"]:
        print("该读者没有借阅这本书的记录。")
        return

    # 移除借阅记录
    book["borrowed_by"] = [entry for entry in book["borrowed_by"] if entry["member_id"] != member_id]
    book["available_copies"] += 1
    member["borrowed"].remove(isbn)
    save_data(data)
    print(f"归还成功！《{book['title']}》已从 {member_id} 收回。")

def show_borrowed_books(data, member_id):
    member = data["members"].get(member_id)
    if not member or not member["borrowed"]:
        print("该读者当前没有借阅任何图书。")
        return
    print(f"读者 {member_id} 当前借阅：")
    for isbn in member["borrowed"]:
        book = data["books"].get(isbn)
        if book:
            print(f"  {isbn} - 《{book['title']}》")

def main():
    data = load_data()
    print("=== 图书管理系统 ===")
    while True:
        print("\n选项：1.添加图书 2.删除图书 3.更新图书 4.搜索图书 5.借阅 6.归还 7.查看借阅 8.退出")
        choice = input("请输入选项: ").strip()

        if choice == "1":
            isbn = input("ISBN: ").strip()
            title = input("书名: ").strip()
            author = input("作者: ").strip()
            copies = int(input("副本数(默认1): ").strip() or "1")
            add_book(data, isbn, title, author, copies)

        elif choice == "2":
            isbn = input("要删除的ISBN: ").strip()
            remove_book(data, isbn)

        elif choice == "3":
            isbn = input("要更新的ISBN: ").strip()
            print("留空则不修改该项")
            title = input("新书名: ").strip()
            author = input("新作者: ").strip()
            copies = input("新副本数: ").strip()
            update_book(data, isbn, title if title else None,
                        author if author else None,
                        int(copies) if copies else None)

        elif choice == "4":
            keyword = input("输入书名/作者/ISBN关键词: ").strip()
            search_books(data, keyword)

        elif choice == "5":
            member_id = input("借阅人ID: ").strip()
            isbn = input("要借阅的ISBN: ").strip()
            borrow_book(data, member_id, isbn)

        elif choice == "6":
            member_id = input("归还人ID: ").strip()
            isbn = input("归还的ISBN: ").strip()
            return_book(data, member_id, isbn)

        elif choice == "7":
            member_id = input("读者ID: ").strip()
            show_borrowed_books(data, member_id)

        elif choice == "8":
            print("再见！")
            break
        else:
            print("无效选项，请重新输入。")

if __name__ == "__main__":
    main()

**问题理解**  
- 进水口单独注满水池需要 3 小时。  
- 出水口单独排空水池需要 5 小时。  
- 问：两个口同时打开，需要几小时才能注满水池？

---

**第一步：计算进水速率**  
进水 3 小时注满整个水池，所以每小时的注水量是：  
\[
\text{进水速率} = \frac{1}{3} \ \text{池/小时}
\]

**第二步：计算出水速率**  
出水 5 小时排空整个水池，所以每小时的排水量是：  
\[
\text{出水速率} = \frac{1}{5} \ \text{池/小时}
\]

**第三步：计算同时开时的净注水速率**  
同时开时，进水在加水，出水在放水，实际每小时水池增加的水量为两者之差：  
\[
\text{净速率} = \frac{1}{3} - \frac{1}{5}
\]
找到公分母 15：  
\[
\frac{1}{3} = \frac{5}{15}, \quad \frac{1}{5} = \frac{3}{15}
\]
所以：  
\[
\text{净速率} = \frac{5}{15} - \frac{3}{15} = \frac{2}{15} \ \text{池/小时}
\]

**第四步：计算注满所需时间**  
水池总量为 1 个池，每小时净增加 \(\frac{2}{15}\) 池，所需时间就是总量除以净速率：  
\[
\text{时间} = \frac{1}{\frac{2}{15}} = \frac{15}{2} = 7.5 \ \text{小时}
\]

### 通用解法

假设总共有 **H** 个头，**F** 只脚。  
鸡有 2 只脚，兔有 4 只脚。

#### 方法一：假设全是鸡（算术法）
1. **假设全部都是鸡**：那么脚的总数应该是 \( 2 \times H \)。
2. **计算脚数差**：实际脚数比假设脚数多出 \( F - 2H \) 只脚。
3. **求兔数量**：每只兔比鸡多 \( 4 - 2 = 2 \) 只脚，所以多出的脚数除以 2 就是兔的数量。  
   \[
   \text{兔数} = \frac{F - 2H}{2}
   \]
4. **求鸡数量**：总头数减去兔数。  
   \[
   \text{鸡数} = H - \text{兔数}
   \]

#### 方法二：假设全是兔（算术法）
1. **假设全部都是兔**：那么脚的总数应该是 \( 4 \times H \)。
2. **计算脚数差**：假设脚数比实际脚数多出 \( 4H - F \) 只脚。
3. **求鸡数量**：多出的脚数是因为把鸡都当成兔了，每只鸡被多算了 2 只脚，所以除以 2 就是鸡的数量。  
   \[
   \text{鸡数} = \frac{4H - F}{2}
   \]
4. **求兔数量**：总头数减去鸡数。  
   \[
   \text{兔数} = H - \text{鸡数}
   \]

#### 方法三：方程法
设鸡有 \( x \) 只，兔有 \( y \) 只，列方程组：
\[
\begin{cases}
x + y = H \\
2x + 4y = F
\end{cases}
\]
将第一个方程乘以 2 得 \( 2x + 2y = 2H \)，与第二个方程相减消去 \( x \)：
\[
(2x + 4y) - (2x + 2y) = F - 2H \quad\Rightarrow\quad 2y = F - 2H
\]
所以 \( y = \dfrac{F - 2H}{2} \)，结果与假设全是鸡的方法一致。

## 1. 引言

在过去的十年间，人工智能完成了从“弱人工智能”向“生成式人工智能”的跨越。如今，智能推荐决定着我们阅读什么新闻，算法模型辅助法官判断罪犯的再犯风险，大语言模型承担起教育辅导与情感陪伴的角色。我们正生活在一个被德国社会学家贝克所言的“风险社会”的升级版本中——一种由算法驱动的新型风险社会。

学界与工业界过度聚焦于AI的精准度与效率，却在很大程度上忽视了技术在嵌入社会结构时产生的摩擦与破坏。当AI系统开始介入人类的高阶认知活动时，其对人类自主性、社会公平性以及心理健康的侵蚀便不再是科幻小说的情节，而是迫在眉睫的现实。本文将剥离技术乐观主义的滤镜，深度审视AI在日常生活中引发的五重危害。

## 2. 隐私边界的消融：从数据收集到全息画像

### 2.1 被动监控与深度伪造
在传统的互联网时代，隐私泄露多表现为个人身份信息的盗用。而在AI时代，隐私危机体现为对人格权的深度入侵。通过遍布日常的智能音箱、智能穿戴设备及手机传感器，AI不仅收集语音、心率、步态，还能通过麦克风捕捉环境音，推断用户是否处于独处或对话状态。
更为严峻的是生成式对抗网络（GAN）与扩散模型（Diffusion Models）催生的深度伪造（Deepfake）技术。仅需几张公开社交照片，恶意使用者即可将任何人的面部移植至色情视频或犯罪现场证据中。这种危害并非仅仅是名誉损害，而是导致个体对其数字身份的“失控”，造成难以修复的社会性死亡。

### 2.2 群体隐私的遮蔽
传统隐私保护框架基于个体同意原则，但AI的推理能力使得“群体隐私”遭受冲击。大数据分析可从群体行为模式中推断出敏感属性。例如，某用户的种族、健康状态甚至性取向，往往并非由其本人泄露，而是通过其社交网络好友的特征向量聚类推算得出。在这种情境下，个体即使未参与任何数据上传，也难以独善其身。这种基于关联规则的透视，将全社会笼罩在一个无形的“圆形监狱”之中。

## 3. 认知能力的退化：算法调度下的思维窄化

### 3.1 信息茧房的自动化加固
推荐算法以“千人千面”之名，行“思想圈禁”之实。协同过滤与内容推荐算法通过最大化用户停留时长，不断投喂高唤醒度、高同质化的内容。这种机制导致了个体认知极化的加剧。CAS（复杂适应系统）理论指出，长期的同质化信息输入不仅会让个体陷入确认偏差，更会削弱其面对复杂现实的抗逆力。
在大语言模型（LLM）普及后，这一危害进一步加重。当人们习惯于向ChatGPT这类工具寻求答案时，知识获取过程从“检索、筛选、批判性思考”退化为“提问、接受”。这种认知卸载虽提升了短期效率，却可能导致人类长时记忆萎缩、深度阅读能力衰退及辩证思维能力的普遍退化。

### 3.2 创造力的同质化陷阱
生成式AI被标榜为激发灵感的工具，但其生成逻辑本质上是基于海量数据概率分布的“均质化重组”。当作家依赖AI续写、设计师依赖AI生成初稿时，作品往往趋向于统计学上的“平均审美”。这种由最大似然估计驱动的输出，滤掉了那些边缘、非主流但极具突破性的怪诞灵感。日常生活中的艺术表达正在沦为算法的复制品，丧失本雅明笔下的“灵韵”。

## 4. 算法歧视：自动化下的结构性暴力

### 4.1 偏见的数据化闭环
AI并非天生邪恶，但它忠实地复现并放大了训练数据中的历史偏见。在日常生活场景中，这种歧视变得愈发隐蔽且难以辩驳。
例如，在智能招聘系统中，若历史数据中高级程序员多为男性，算法会自然习得“男性”与“资深”的虚假关联，在简历筛选中直接过滤女性求职者。在医疗领域，由于少数族裔的临床数据在历史记录中占比较低，AI诊断系统在皮肤癌或肾病检测中对深色肤色人群的误诊率显著偏高。
这种歧视之所以构成巨大的结构暴力，在于其“数学洗白”的特性。受害者很难反抗一个“客观中立的数学公式”，而算法的规模化应用使得歧视不再是偶然的个体行为，而成为系统性的排斥机制。

### 4.2 预测性警务与贫困陷阱
在美国、英国等部分地区，预测性警务系统被用于指导警力部署。由于少数族裔社区的逮捕率历史数据偏高，算法会将这类社区标记为高风险区，导致更多警察巡逻，进而产生更多的轻微罪行逮捕记录，再次喂入算法。这种正反馈循环构成了一种现代版的“数字猎巫”，将贫困与种族固化为原罪，剥夺了弱势群体日常生活的安全感。

## 5. 社会关系的异化：从人际信任到机器中介

### 5.1 情感计算与陪伴剥削
随着孤独经济的兴起，AI伴侣应用层出不穷。这些AI被设计得永不失去耐心、永远提供正面反馈且无条件服从。这种完美的人机关系实质上扭曲了人类的依恋模式。人与人的交往本就伴随着冲突、协商与妥协，这是社会性人格形成的基础。而AI陪伴提供了一种排他性的、无摩擦的交流环境，导致个体在回归现实人际网络时，对正常的社交摩擦容忍度极低。这便是“技术驯化”下的亲密关系异化。

### 5.2 责任的弥散与道德洗白
在决策系统中，AI的介入导致责任的弥散。当自动驾驶汽车发生事故、医疗AI误诊或信贷AI错误拒贷时，责任主体变得模糊不清：是开发者的代码缺陷？是训练数据的责任？还是使用者未及时干预？这种“有组织的不负责任”让受害者陷入维权黑洞。甚至，人类开始利用AI作为道德挡箭牌，例如：客服系统中的“算法规定无法退换”，实则是企业利用机器决定论来推卸本应承担的人文关怀责任。

## 6. 安全与失控：当基础设施遭遇“幻觉”

### 6.1 大模型的“幻觉”灾难
生成式AI的“幻觉”特性，即自信满满地编造看似合理实则错误的事实，是其底层技术的固有缺陷。当这一特性渗透至日常生活中，便构成了严重的信息污染。在医疗咨询中，AI可能推荐错误的药物剂量；在法律咨询中，AI可能杜撰不存在的判例。这种危害在家庭教育中尤为致命，儿童尚未建立完善的知识验证体系，极易将AI的幻觉内化为真实认知，造成难以纠正的教育灾难。

### 6.2 自动化紧耦合与系统性崩塌
现代社会的基础设施如水、电、交通、金融，已深度依赖AI调度。这种紧耦合系统在流形混沌状态下极其脆弱。看似微小的训练数据偏差或对抗性样本攻击，可能导致全城停电、股市闪崩。2024年7月发生的“微软蓝屏”事件虽非纯AI故障，却警示我们：当一切都交给自动化的中央神经系统，人类便丧失了在危机时刻手动接管系统的能力与直觉。这种对复杂性的失控，是AI嵌入物理世界最深层的隐患。

## 7. 深层逻辑：工具理性对生活世界的殖民

上述五重危害的背后，是哈贝马斯所警告的“系统对生活世界的殖民”在数字时代的终极体现。AI技术在资本与权力的裹挟下，体现出极强的工具理性特征：

1.  **量化一切的冲动**：AI无法理解无法被量化的价值，如尊严、情感、直觉。当生活被AI全方位接入，一切行为都被迫化为可计算的参数，人类存在的丰富性与模糊性被削足适履。
2.  **对效率的极致崇拜**：AI看似节省了思考与选择的时间，实则剥夺了人类在犹豫、犯错和反思中成长的自由。以牺牲自由意志换取效率，构成了现代科技人文主义特有的焦虑。
3.  **技术解决方案主义**：社会试图用技术手段解决技术带来的问题，却回避了深层的制度缺陷与分配不公。

## 8. 结论与进路：走向节制与对齐

人工智能在日常生活中的危害并非技术的必然宿命，而是技术发展缺乏伦理规约的症候。面对这些危害，简单的反技术主义或卢德主义并不可取，我们需要建立一种更具韧性的共存策略：

1.  **可解释性强制**：在高风险领域（如医疗、司法、信贷），应立法强制AI系统具备可解释性，禁止不可解释的黑箱模型左右人类的基本权利。受害者有权要求算法决策的逻辑复核。
2.  **数据去偏与对抗训练**：技术上需从数据预处理、模型训练阶段的对抗去偏算法入手，消除对特定群体的刻板印象。但这仅仅是治标，治本在于人类社会的实质性平权。
3.  **数字断连权与认知留白**：社会应保障个体的“不被迫使用AI”的权利。在公共服务中保留人工通道，在教育领域强调计算思维与批判性思维的同步培养，确保人类认知能力不退场。
4.  **人机回环的伦理审查**：建立跨学科的伦理委员会，不仅审查技术指标，更审查AI应用对弱势群体心理、社会结构影响的长期效应。

AI像一面镜子，映照的是人类自身的理性与疯癫。只有当人类学会正视自身社会的结构性缺陷，不再试图用算法来粉饰不公时，AI才能从一个潜在的危害源，回归到赋能人类生活的本体位置。
